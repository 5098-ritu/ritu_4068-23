Using %TYPE in PL/SQL


            
Syntax:

            DECLARE
                var_name table_name.column_name%TYPE;
            BEGIN
                -- Your PL/SQL logic here
            END;
            
            
Ex:

            DECLARE 
                v_name Students.Name%TYPE;
                v_courseid Courses.CourseID%TYPE;
            BEGIN
                SELECT Name, CourseID INTO v_name, v_courseid
                FROM Students
                WHERE CourseID = 'C1';
                DBMS_OUTPUT.PUT_LINE('Student: ' || v_name || ', Course ID: ' || v_courseid);
            END;

Output

Table Before Execution:


            +--------+----------+-----------+
            | Name   | CourseID | CourseName|
            +--------+----------+-----------+
            | Lucky  | C1       | Math      |
            | Ajay   | C2       | Science   |
            | Mahi   | C3       | History   |
            | Neha   | C4       | English   |
            +--------+----------+-----------+
            4 rows in set (0.00 sec)

PL/SQL Execution Result (via Oracle Live SQL):


            Student: Lucky, Course ID: C1

Named Function Example


            
Syntax:

            CREATE OR REPLACE FUNCTION function_name (parameter1 datatype)
            RETURN return_datatype
            IS
            BEGIN
                RETURN result;  -- Return the result
            END;

Creating and Using a Stored Function in PL/SQL


                    
 A stored function is a function that is saved in the database and can be called multiple times, either from SQL or other PL/SQL blocks.

            
Syntax:

            CREATE OR REPLACE FUNCTION function_name (parameter1 datatype)
            RETURN return_datatype
            IS
            BEGIN
                -- Function logic
                RETURN result;  -- Return the result
            END;
            
            
Ex:

            CREATE OR REPLACE FUNCTION get_square (num IN NUMBER)
            RETURN NUMBER
            IS
            BEGIN
                RETURN num * num;  -- Return the square of the number
            END;
            
            -- Example of calling the stored function from SQL or PL/SQL:
            DECLARE
                num NUMBER := 5;
                result NUMBER;
            BEGIN
                result := get_square(num);  -- Calling the stored function
                DBMS_OUTPUT.PUT_LINE('Square of ' || num || ' is: ' || result);  -- Output the result
            END;

Output

Stored Function Before Execution:


            CREATE OR REPLACE FUNCTION get_square (num IN NUMBER)
            RETURN NUMBER
            IS
            BEGIN
                RETURN num * num;  -- Return the square of the number
            END;

PL/SQL Execution Result (via Oracle Live SQL):


            Square of 5 is: 25

Exception Handling with Predefined Exceptions


            
Syntax:

            BEGIN
               -- PL/SQL statements that may raise predefined exceptions
            
            EXCEPTION
               WHEN predefined_exception1 THEN
                  -- Handling code for predefined_exception1
            
               WHEN predefined_exception2 THEN
                  -- Handling code for predefined_exception2
            
               -- Add more predefined exceptions as needed
            
               WHEN OTHERS THEN
                  -- Generic handler for all other exceptions
            
            END;
            
Ex:

            DECLARE
               RESULT  NUMBER;
               NAME_S  VARCHAR2(50);
            
            BEGIN
               SELECT NAME INTO NAME_S FROM STUDENT WHERE ROLLNO = 9999; -- MAY RAISE NO_DATA_FOUND
            
               RESULT := 10 / 0;  -- MAY RAISE ZERO_DIVIDE
            
               DBMS_OUTPUT.PUT_LINE('RESULT: ' || RESULT);
            
            EXCEPTION
            
               WHEN NO_DATA_FOUND THEN
                  DBMS_OUTPUT.PUT_LINE('NO STUDENT FOUND WITH THAT ROLL NUMBER.');
            
               WHEN ZERO_DIVIDE THEN
                  DBMS_OUTPUT.PUT_LINE('DIVISION BY ZERO ERROR.');
            
               WHEN OTHERS THEN
                  DBMS_OUTPUT.PUT_LINE('AN UNEXPECTED ERROR OCCURRED: ' || SQLERRM);
            END;

Output


            Statement processed.
            No student found with that roll number.

User-Defined Exception


            
Syntax:

            DECLARE
              -- Declare a user-defined exception
                 exception_name EXCEPTION;
            
              -- Optional: associate it with an error number using PRAGMA (advanced)
              -- PRAGMA EXCEPTION_INIT(exception_name, error_number);
            
            BEGIN
                IF some_condition THEN
                RAISE exception_name;  -- Raise the user-defined exception
                END IF;
            
            EXCEPTION
                WHEN exception_name THEN
                DBMS_OUTPUT.PUT_LINE('Custom exception occurred');
            END;
            
            
Ex:

            DECLARE
              my_exception EXCEPTION;
              num NUMBER := -10;
            BEGIN
            IF num < 0 THEN
                RAISE my_exception;
            END IF;
            
            EXCEPTION
                WHEN my_exception THEN
                DBMS_OUTPUT.PUT_LINE('Negative number is not allowed!');
            END;

Output


            Statement processed.
            Negative number is not allowed!

Variable (Accessing)



Syntax:

DECLARE
    variable_name data_type := initial_value;
BEGIN
    -- Access the variable by referencing it directly
    DBMS_OUTPUT.PUT_LINE(variable_name);  -- Access and print the value
END;


Ex:

1: Accessing Variables in an Anonymous Block
DECLARE
    v_name VARCHAR2(50) := 'Alice';  -- Declare and initialize variable
    v_age  NUMBER := 30;             -- Declare and initialize variable
BEGIN
    -- Access and print variables
    DBMS_OUTPUT.PUT_LINE('Name: ' || v_name);  -- Accessing v_name
    DBMS_OUTPUT.PUT_LINE('Age: ' || v_age);   -- Accessing v_age
END;

2: Using Variables in Calculations
DECLARE
    v_price   NUMBER := 100;      -- Initialize product price
    v_discount NUMBER := 20;      -- Initialize discount percentage
    v_final_price NUMBER;         -- Declare variable to store final price
BEGIN
    -- Accessing variables and performing calculations
    v_final_price := v_price - (v_price * v_discount / 100);  -- Calculating discounted price

    DBMS_OUTPUT.PUT_LINE('Original Price: ' || v_price);        -- Access and print original price
    DBMS_OUTPUT.PUT_LINE('Discount: ' || v_discount || '%');     -- Access and print discount
    DBMS_OUTPUT.PUT_LINE('Final Price: ' || v_final_price);      -- Access and print final price
END;

Output


1: Accessing Variables in an Anonymous Block
Statement processed.
Name: Alice
Age: 30

2: Using Variables in Calculations
Statement processed.
Original Price: 100
Discount: 20%
Final Price: 80

Variable (Scope and Visibility)



Syntax:

Example of Local Variable Scope:
DECLARE
    v_name VARCHAR2(50); -- Local variable declaration
BEGIN
    v_name := 'Alice';  -- Local variable assignment
    DBMS_OUTPUT.PUT_LINE('Name: ' || v_name);
END;

Example of Package-Level Variable Scope:
CREATE OR REPLACE PACKAGE my_package IS
    v_global_name VARCHAR2(50);  -- Global variable in the package
    PROCEDURE print_name;
END my_package;

Example of Variable Visibility (Within a Block):
DECLARE
    v_local_name VARCHAR2(50); -- Declared in the DECLARE section
BEGIN
    v_local_name := 'Bob';  -- Initialized inside the BEGIN block
    DBMS_OUTPUT.PUT_LINE('Name: ' || v_local_name);  -- Visible within this block
END;


Ex:

Example of Shadowing (Same Variable Name with Different Scope)

DECLARE
    v_value NUMBER := 10;  -- Outer scope variable
BEGIN
    DBMS_OUTPUT.PUT_LINE('Outer v_value: ' || v_value);  -- Prints 10

    DECLARE
        v_value NUMBER := 20;  -- Inner scope variable, shadows outer v_value
    BEGIN
        DBMS_OUTPUT.PUT_LINE('Inner v_value: ' || v_value);  -- Prints 20
    END;

    DBMS_OUTPUT.PUT_LINE('Outer v_value again: ' || v_value);  -- Prints 10
END;

Output


Outer v_value: 10
Inner v_value: 20
Outer v_value again: 10

Explicit Cursor



Syntax:

DECLARE
    -- Declare cursor
    CURSOR cursor_name IS
        SELECT column1, column2 FROM table_name WHERE condition;

    -- Declare variables to hold data from the cursor
    v_column1 table_name.column1%TYPE;
    v_column2 table_name.column2%TYPE;
BEGIN
    -- Open the cursor
    OPEN cursor_name;

    -- Loop through the cursor and fetch rows
    LOOP
        FETCH cursor_name INTO v_column1, v_column2;
        
        -- Exit loop when no more rows to fetch
        EXIT WHEN cursor_name%NOTFOUND;

        -- Process each row (e.g., print values)
        DBMS_OUTPUT.PUT_LINE('Column 1: ' || v_column1 || ', Column 2: ' || v_column2);
    END LOOP;

    -- Close the cursor
    CLOSE cursor_name;
END;


Ex:

DECLARE
    -- Declare the cursor
    CURSOR emp_cursor IS
        SELECT first_name, last_name, salary FROM employees;
    
    -- Declare variables to hold employee data
    v_first_name employees.first_name%TYPE;
    v_last_name employees.last_name%TYPE;
    v_salary employees.salary%TYPE;
BEGIN
    -- Open the cursor
    OPEN emp_cursor;
    
    -- Loop through each row of the cursor
    LOOP
        -- Fetch the data into variables
        FETCH emp_cursor INTO v_first_name, v_last_name, v_salary;

        -- Exit loop if no more rows
        EXIT WHEN emp_cursor%NOTFOUND;

        -- Display the employee information
        DBMS_OUTPUT.PUT_LINE('Employee: ' || v_first_name || ' ' || v_last_name || ', Salary: ' || v_salary);
    END LOOP;

    -- Close the cursor
    CLOSE emp_cursor;
END;

Output


Employee: Steven King, Salary: 24000
Employee: Neena Kochhar, Salary: 17000
Employee: Lex De Haan, Salary: 17000

Trigger Syntax: CREATE



Syntax:

CREATE [OR REPLACE] TRIGGER trigger_name
{BEFORE | AFTER | INSTEAD OF}
{INSERT | UPDATE | DELETE}
ON table_name
[FOR EACH ROW]
DECLARE
   -- variable declarations (optional)
BEGIN
   -- trigger logic
END;


Ex:

CREATE OR REPLACE TRIGGER trg_after_insert
AFTER INSERT
ON students
FOR EACH ROW
BEGIN
   DBMS_OUTPUT.PUT_LINE('A new student was inserted with Roll No: ' || :NEW.rollno);
END;

Output


Trigger created.


Execution

VALUES (4106,17,'RIYA');

A new student was inserted with Roll No: 17

BEFORE INSERT Trigger Example



Syntax:

CREATE OR REPLACE TRIGGER trigger_name
BEFORE INSERT ON table_name
FOR EACH ROW
BEGIN
   -- Logic before inserting a row
END;


Ex:

CREATE OR REPLACE TRIGGER before_insert_trigger
BEFORE INSERT ON student
FOR EACH ROW
BEGIN
   DBMS_OUTPUT.PUT_LINE('Before inserting: ' || :NEW.name);
END;

Output


Trigger created.


Execution

INSERT INTO student (rollno, name, marks) VALUES (1, 'Lucky', 90);


Output

Before inserting: Lucky

Query

BEFORE UPDATE Trigger Syntax



Syntax:

CREATE OR REPLACE TRIGGER trigger_name
BEFORE UPDATE ON table_name
FOR EACH ROW
BEGIN
   -- Logic before updating a row
END;


Ex:

CREATE OR REPLACE TRIGGER before_update_trigger
BEFORE UPDATE ON student
FOR EACH ROW
BEGIN
   DBMS_OUTPUT.PUT_LINE('Updating name from ' || :OLD.name || ' to ' || :NEW.name);
END;

Output


Trigger created.  


Execution

UPDATE student SET name = 'Ajay' WHERE rollno = 1;


Output

Updating name from Lucky to Ajay

AFTER INSERT Trigger Syntax



Syntax:

CREATE OR REPLACE TRIGGER trigger_name
AFTER INSERT ON table_name
FOR EACH ROW
BEGIN
   -- Logic after inserting a row
END;


Ex:

CREATE OR REPLACE TRIGGER after_insert_trigger
AFTER INSERT ON student
FOR EACH ROW
BEGIN
   DBMS_OUTPUT.PUT_LINE('Student inserted: ' || :NEW.name);
END;

Output


Trigger created.  


Execution

INSERT INTO student (rollno, name, marks) VALUES (2, 'Neha', 88);


Output

Student inserted: Neha

Drop Trigger Syntax



Syntax:

DROP TRIGGER trigger_name;


Ex:

Output

View User Triggers Syntax



Syntax:

SELECT * FROM USER_TRIGGERS;
or (to select specific columns):
SELECT trigger_name, table_name, triggering_event, trigger_type, status 
FROM user_triggers;

SELECT trigger_name, table_name, triggering_event, trigger_type, status 
FROM user_triggers;
